# Project Structure

```
src/
  server.py          # FastMCP server — tool definitions and entry point
  usage_logger.py    # Decorator that logs tool calls to logs/usage.jsonl
  validators.py      # Input validation for dates, model names, account IDs, data dir
  models/            # Dataclasses: Account, Holding, Transaction
  loaders/
    base.py          # BaseLoader ABC
    csv_loader.py    # Reads accounts.csv, transactions.csv, holdings.csv
    plaid_loader.py  # Plaid API loader (alternative data source, stub)
  calculators/
    net_worth.py     # Assets minus liabilities
    budget.py        # Budget breakdown logic
    budget_models.py # Budget model definitions (50/30/20, zero-based, etc.)
    performance.py   # Brokerage gain/loss and allocation
    savings_rate.py  # Income vs. savings/retirement contributions
    spending_trends.py # Month-over-month expense trends by category
tests/               # pytest test suite mirroring src/ structure
.claude/
  agents/            # Agent definition files (.md) for the team
  commands/          # Custom slash commands (Markdown files)
  context/           # Context files loaded into conversations
  outputs/           # Files generated by Claude (analyses, reports, etc.)
  skills/            # Custom agent skill definitions
```
